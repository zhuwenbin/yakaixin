```dart
Future<void> retryFailedReceipts() async {
  final failedReceipts = await IAPReceiptCache.getFailedReceipts();
  
  for (final receipt in failedReceipts) {

        // ✅ 更新重试次数

      }
    } catch (e) {
      await IAPReceiptCache.updateRetryCount(receipt['orderId']);
    }
  }
}
```

```dart
// 建议：动态调整清理超时时间
Future<void> _clearPendingTransactions({
  Duration timeout = const Duration(seconds: 3),
  int retryCount = 0,
}) async {
  try {
    _isCleaningMode = true;
    _cleaningCompleter = Completer<void>();
    
    await _iap.restorePurchases();
    
    try {
      await _cleaningCompleter!.future.timeout(timeout);
    } catch (e) {
      debugPrint('清理超时，可能没有待处理交易');
      
      // ✅ 建议：重试清理
      if (retryCount < 2) {
        debugPrint('🔄 重试清理待处理交易...');
        await Future.delayed(const Duration(milliseconds: 500));
        return _clearPendingTransactions(
          timeout: Duration(seconds: timeout.inSeconds + 2),
          retryCount: retryCount + 1,
        );
      }
    }
  } finally {
    _isCleaningMode = false;
    _cleaningCompleter = null;
  }
}
```

```dart
// 建议：错误分类处理
enum IAPErrorType {
  networkError,      // 网络错误
  verificationError, // 验证错误
  duplicateError,    // 重复购买错误
  canceledError,     // 用户取消
  unknownError,      // 未知错误
}

IAPErrorType _classifyError(dynamic error) {
  if (error.toString().contains('storekit_duplicate_product_object')) {
    return IAPErrorType.duplicateError;
  }
  if (error.toString().contains('network') || error.toString().contains('timeout')) {
    return IAPErrorType.networkError;
  }
  if (error.toString().contains('canceled')) {
    return IAPErrorType.canceledError;
  }
  return IAPErrorType.unknownError;
}

// 根据错误类型提供不同的处理策略
String _getErrorMessage(IAPErrorType errorType) {
  switch (errorType) {
    case IAPErrorType.networkError:
      return '网络连接失败，请检查网络后重试';
    case IAPErrorType.verificationError:
      return '验证失败，已保存数据，将自动重试';
    case IAPErrorType.duplicateError:
      return '检测到未完成的交易，正在处理...';
    case IAPErrorType.canceledError:
      return '您已取消支付';
    case IAPErrorType.unknownError:
      return '支付失败，请稍后重试';
  }
}
```
